En esta versin del cdigo, se ha eliminado la funcin countRows ya que se puede utilizar directamente x.shape para obtener el nmero de filas y columnas de la matriz de entrada x.

Adems, se ha reemplazado las operaciones mm por torch.matmul para realizar las multiplicaciones de matrices.

La inicializacin de los pesos y sesgos se realiza mediante las funciones randomWeights e initializeBias, respectivamente, para generar valores aleatorios y ceros.

El clculo de la salida de la red neuronal se realiza mediante la funcin forward, que realiza las operaciones de multiplicacin de matrices y suma de sesgos utilizando la funcin de activacin sigmoide.

Finalmente, el resultado se transpone para que tenga la forma correcta y se muestra el grfico de la red neuronal utilizando plt.scatter.

import torch
import numpy as np
from matplotlib import pyplot as plt
from matplotlib import cm

def sigmoidalActivationFunction(x):
    return 1 / (1 + torch.exp(-x))

def forward(weights1, weights2, featuresT, bias1, bias2):
    hidden1 = sigmoidalActivationFunction(torch.matmul(weights1, featuresT) + bias1)
    hidden2 = sigmoidalActivationFunction(torch.matmul(weights2, hidden1) + bias2)
    return hidden2

def randomWeights(rows, columns):
    return torch.rand(rows, columns)

def initializeBias(rows):
    return torch.zeros(rows, 1)

# Parmetros de la red neuronal
numObservations = 3
numFeatures = 2
countNeuronsHiddenLayer1 = 2
countNeuronsHiddenLayer2 = 2

# Datos de entrada
x = torch.tensor([[0.05, 0.01], [0.02, 0.5], [2, 4]])
rows, columns = x.shape

# Generar pesos aleatorios y sesgos
weights1 = randomWeights(countNeuronsHiddenLayer1, numFeatures)
weights2 = randomWeights(countNeuronsHiddenLayer2, countNeuronsHiddenLayer1)
bias1 = initializeBias(countNeuronsHiddenLayer1)
bias2 = initializeBias(countNeuronsHiddenLayer2)

# Transponer las caractersticas
featuresT = x.transpose(0, 1)

# Calcular la salida de la red neuronal
hidden1 = sigmoidalActivationFunction(torch.matmul(weights1, featuresT) + bias1)
hidden2 = sigmoidalActivationFunction(torch.matmul(weights2, hidden1) + bias2)

output = hidden2[:2].transpose(0, 1)  # Utilizar solo las primeras dos columnas de la salida
output_np = output.detach().numpy()  # Convertir a un arreglo NumPy

# Mapear los valores de salida a un solo color utilizando el mapa de colores
norm = plt.Normalize(output_np.min(), output_np.max())
color = cm.coolwarm(norm(output_np.mean()))

# Dibujar la red neuronal
plt.figure(figsize=(6, 6))
plt.scatter(x[:, 0], x[:, 1], color=color)
plt.xlabel('Feature 1')
plt.ylabel('Feature 2')
plt.title('Red Neuronal con Capa Oculta 1 y Capa Oculta 2')
plt.colorbar(label='Salida')
plt.show()

import torch

def sigmoidalActivationFunction(x):
    return 1 / (1 + torch.exp(-x))

def forward(weights, featuresT):
    result = weights.mm(featuresT)
    return sigmoidalActivationFunction(result)

def calculateError(output, target):
    error = target - output
    return error

x = torch.tensor([[0.05, 0.01], [0.02, 0.5], [2, 4]])
target = torch.tensor([0, 1, 1])

countNeuronsHiddenLayer1 = 1
weights = torch.rand(countNeuronsHiddenLayer1, x.size()[1])
featuresT = x.transpose(0, 1)

output = forward(weights, featuresT)
error = calculateError(output, target)

print("Output:", output)
print("Error:", error)
